Const & Let

ES6 
var a = 0; // standart varable
function f() {
 {
   let x;
   {
     // okay, block scoped name
     const x = "sneaky";
     // error, const
     x = "foo";
   }
   // error, already declared in block
   let x = "inner";
  }
}
